def test_x_axis_labels_1():
    '''
    X Axis Labels 1
    
    Simple X Axis Lables
    '''
    t = title(text='X Axis Labels')
    l = line_dot()
    l.values = range(9,0,-1)
    
    x = x_axis()
    x.colour = '#428C3E'
    x.grid_colour = '#86BF83'
    lbl = labels(labels=['one', 'two', 'three', 'four', 'five', 'six', 'seven', 'eight', 'nine'])
    x.labels = lbl
    chart = open_flash_chart()
    chart.title = t
    chart.add_element(l)
    chart.x_axis = x
    return chart
