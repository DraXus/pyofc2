Example of Multiple Line Charts

def test_manylines():
           
    l1 = line_dot()
    l1.width = 4
    l1.colour = "#DFC329"
    l1.dot_size = 5
    l1.values = [2,4,1,3,5,6,8,9,7]
    l1.text = 'Line 1'
    l1.font_size = 10
    
    l2 = line_hollow()
    l2.width = 1
    l2.colour = "#6363ac"
    l2.dot_size = 5
    l2.values = [8,7,10,9,11,13,12,9,7,8]  
    l2.text = 'Line 2'
    l2.font_size = 10
    
    l3 = line()
    l3.width = 4;
    l3.colour = "#5E4725"
    l3.values = [15,14,19,16,17,18,15,16,14] 
    l3.text = 'Line 3'
    l3.font_size = 10
    
    y = y_axis()
    y.min, y.max, y.steps = 0, 20, 5
    
    t = title(text='Three lines example') 
    chart = open_flash_chart()
    chart.title = t
    chart.add_element(l1)
    chart.add_element(l2)    
    chart.add_element(l3)
    chart.y_axis = y
    return chart
